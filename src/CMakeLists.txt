# Geary client app
#################################################

    ${CLIENT_SRC}
PACKAGES
    ${CLIENT_PACKAGES}
    ${ENGINE_PACKAGES}
CUSTOM_VAPIS
    "${CMAKE_BINARY_DIR}/src/geary-static.vapi"

    ${VALAC_OPTIONS}
    --vapidir=${CMAKE_BINARY_DIR}/src
)

target_link_libraries(geary ${DEPS_LIBRARIES} gthread-2.0 geary-static $ENV{HOME}/webkit/lib/libwebkitgtk-3.0.so $ENV{HOME}/webkit/lib/libsoup-2.4.so $ENV{HOME}/webkit/lib/libgio-2.0.so)

    TARGET
        geary
    POST_BUILD
    COMMAND 
        ${CMAKE_COMMAND} -E copy geary ${CMAKE_BINARY_DIR}/
